Query
-
```postgresql
SELECT COUNT(*) FROM Н_ОТДЕЛЫ 
    JOIN Н_ПЛАНЫ ON Н_ПЛАНЫ.ОТД_ИД = Н_ОТДЕЛЫ.ИД
    JOIN Н_УЧЕНИКИ ON Н_ПЛАНЫ.ИД = Н_УЧЕНИКИ.ПЛАН_ИД
    JOIN Н_ЛЮДИ ON Н_УЧЕНИКИ.ЧЛВК_ИД = Н_ЛЮДИ.ИД
    WHERE Н_ОТДЕЛЫ.КОРОТКОЕ_ИМЯ = 'КТиУ' 
      AND Н_УЧЕНИКИ.КОНЕЦ > now() 
      AND Н_УЧЕНИКИ.СОСТОЯНИЕ NOT IN ('отчислен') 
      AND date_part('year', age(Н_ЛЮДИ.ДАТА_РОЖДЕНИЯ)) > 25;
```
