```postgresql
SELECT НОМЕР FROM (
    SELECT Н_ПЛАНЫ.ИД AS НОМЕР, COUNT(*) AS КОЛВО_ГРУПП, ОТД_ИД_ЗАКРЕПЛЕН_ЗА AS ОТД FROM Н_ГРУППЫ_ПЛАНОВ
    JOIN Н_ПЛАНЫ ON Н_ПЛАНЫ.ИД = Н_ГРУППЫ_ПЛАНОВ.ПЛАН_ИД
    GROUP BY (Н_ПЛАНЫ.ИД, ОТД_ИД_ЗАКРЕПЛЕН_ЗА)
) AS ПЛАНЫ
WHERE ОТД = 102 AND КОЛВО_ГРУПП = 2;
```
